void setDarkModeUserConfig(bool value) async {
  String urlDarkModeExtension = "users/configs/darkMode";
  try {
    http.post(url + urlDarkModeExtension,
        headers: {
          "Content-Type": "application/json",
          "Authorization": "Bearer $jwtToken"
        },
        body: jsonEncode({
          "value": "$value",
        }));
  } on Exception {}
}
